def.Func.TeleportPlayer()
if.Player.IsHolding($%defaulttpItem)

{
if.Player.NoItemCooldown($%defaulttpItem)
{
var.=(~vector,gv("Direction"))
var.MultiplyVector(~vector,$%defaultplayerVelocityMulti)
action.Player.SetVelocity(~vector,addtoCurrentVelocity="False")
action.Player.Particle(particle("Portal",amount=50,horizontal=1.5,vertical=2),gv("Location"))
action.Player.SetItemCooldown($%defaulttpItem,$%defaultplayerTeleportCoolDown)
}
}

event.Player.RightClick
call.Func.TeleportPlayer()
if.Player.IsHolding(item("minecraft:tube_coral",name="&ePlay",lore="&7Right Click To Teleport To Arena"))#other item clicks
{
    action.Player.Teleport(location(2.5,60.2,2.5,0,325)) # teleport them to the spawn platform
    action.Player.RemoveItems(item("minecraft:tube_coral",name="&ePlay",lore="&7Right Click To Teleport To Arena"))
}
if.Player.HasSlotItem(9,$%defaulttpItem,)
{
if.Player.NameEquals("olim88")
{
call.Func.CleanBlocks()
call.Func.GenerateArena()
}
}
